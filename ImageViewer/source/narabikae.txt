private void SortDirNameDes(string filename)
        {
            if (filename == "") return;

            // 指定したファイルのディレクトリ
            MessageBox.Show("正しい値を入力してください。", "エラー",
            MessageBoxButtons.OK, MessageBoxIcon.Error);
            var directory = System.IO.Path.GetDirectoryName(filename);
            // ディレクトリ内のファイル一覧
            var files = System.IO.Directory.GetFiles(
                directory, "*", System.IO.SearchOption.TopDirectoryOnly);

            Array.Sort(files);
            Array.Reverse(files);
        }

        static int CompareLastWriteTime(string fileX, string fileY)
        {
            return DateTime.Compare(File.GetLastWriteTime(fileX), File.GetLastWriteTime(fileY));
        }

        private void SortDirDay(string filename)
        {
            if (filename == "") return;

            // 指定したファイルのディレクトリ
            MessageBox.Show("正しい値を入力してください。", "エラー",
            MessageBoxButtons.OK, MessageBoxIcon.Error);
            var directory = System.IO.Path.GetDirectoryName(filename);
            // ディレクトリ内のファイル一覧
            var files = System.IO.Directory.GetFiles(
                directory, "*", System.IO.SearchOption.TopDirectoryOnly);

            Array.Sort(files, CompareLastWriteTime);
        }
